2.5 Identify Function Parameters and Return Values

Functions can take parameters to receive data and return values as output.

2.5.1 Parameters and Arguments

function sum(a, b) {
    return a + b;
}
console.log(sum(10, 5)); // 15

2.5.2 Default Parameters

function greet(name = "Guest") {
    return "Hello, " + name;
}
console.log(greet()); // Hello, Guest

2.5.3 Returning Values

function square(num) {
    return num * num;
}
console.log(square(5)); // 25

Practice Questions

  1. Answer:

    function power(base, exponent = 2) {
    return Math.pow(base, exponent);
}
console.log(power(5)); // Output: 25 (5^2)

  2. Answer:

    function check(num) {
    if (num > 0) return "Positive";
}
console.log(check(-5)); // Output: undefined

  3. Answer:

    function sum(a, b, c) {
    return a + b + c;
}
console.log(sum(1, 2, 3)); // Output: 6
