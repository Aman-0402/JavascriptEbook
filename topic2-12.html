Function Parameters & Return Values

Key Concepts

  • JavaScript uses pass-by-value for primitives
  • JavaScript uses pass-by-reference for objects/arrays
  • To return multiple values, use objects or arrays

Examples

// Pass by Value (Primitives)
let a = 10;
function changeValue(num) {
  num = 20;
}
changeValue(a);
console.log(a); // Output: 10 (unchanged)

// Pass by Reference (Objects)
let person = { name: "Aman" };
function changeName(obj) {
  obj.name = "Raj";
}
changeName(person);
console.log(person.name); // Output: Raj (changed)

Practice Questions

  1. function sum(a, b) {
  return a + b;
}
console.log(sum(5, 10)); // 15
          
  2. let x = 10;
function changeX(num) {
  num = 20;
}
changeX(x);
console.log(x); // Output: 10 (unchanged because numbers are passed by value)
          
  3. let person = { name: "John", age: 30 };
function modifyPerson(obj) {
  obj.age = 35;
}
modifyPerson(person);
console.log(person.age); // Output: 35 (objects are passed by reference)
