3.2 Decision Statements

Explanation

  1. Single Alternative (if) 
    let num = 10;
if (num > 5) {
  console.log("Number is greater than 5");
}
  2. Dual Alternative (if-else) 
    let marks = 75;
if (marks >= 50) {
  console.log("Pass");
} else {
  console.log("Fail");
}
  3. Multiple Alternatives (switch) 
    let day = "Monday";
switch (day) {
  case "Monday":
    console.log("Start of the week!");
    break;
  case "Friday":
    console.log("Weekend is near!");
    break;
  default:
    console.log("Just another day.");
}
  4. Nested if 
    let score = 85;
if (score >= 50) {
  if (score >= 80) {
    console.log("Excellent!");
  } else {
    console.log("Good job!");
  }
} else {
  console.log("Try again!");
}

Practice Questions

  1. let temp = 25;
if (temp > 30) {
  console.log("Hot");
} else if (temp >= 20 && temp <= 30) {
  console.log("Warm");
} else {
  console.log("Cold");
}
// Output: Warm
  2. let a = 5;
let b = 10;
if (a > b) console.log("A is greater");
else if (a == b) console.log("Both are equal");
else console.log("B is greater");
// Output: B is greater
  3. let month = 5; // May
switch (true) {
  case (month >= 3 && month <= 6):
    console.log("Summer");
    break;
  case (month >= 7 && month <= 9):
    console.log("Monsoon");
    break;
  case (month >= 10 || month <= 2):
    console.log("Winter");
    break;
  default:
    console.log("Invalid month");
}
// Output: Summer
