3.3 Loops in JavaScript

Date: 06 April 2025

Types of Loops

1. for loop

for (let i = 1; i <= 5; i++) {
  console.log("Number: " + i);
}

2. while loop

let i = 1;
while (i <= 5) {
  console.log("Count: " + i);
  i++;
}

3. do...while loop

let i = 1;
do {
  console.log("Hello " + i);
  i++;
} while (i <= 3);

4. for...of loop (for arrays)

const fruits = ["apple", "banana", "cherry"];
for (let fruit of fruits) {
  console.log(fruit);
}

5. for...in loop (for objects)

const person = {name: "Aman", age: 22};
for (let key in person) {
  console.log(key + ": " + person[key]);
}

Practice Questions

  1. let i = 1;
while (i <= 10) {
  console.log(i);
  i++;
}
// Output: 1 to 10
  2. const names = ["Aman", "Riya", "Karan"];
for (let name of names) {
  console.log(name);
}
// Output: Aman, Riya, Karan
  3. // for loop: use when number of iterations is known
for (let i = 0; i < 3; i++) {
  console.log("For loop");
}

// while loop: use when condition is checked before each iteration
let j = 0;
while (j < 3) {
  console.log("While loop");
  j++;
}
