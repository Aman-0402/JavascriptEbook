1.4.1 try...catch

try {
    let result = 10 / 0;
    console.log(result);
} catch (error) {
    console.log("An error occurred:", error.message);
}

try {
    let result = 10 / 0;
    if (!isFinite(result)) {
        throw new Error("Division by zero resulted in Infinity!");
    }
    console.log(result);
} catch (error) {
    console.log("An error occurred:", error.message);
}

1.4.2 finally

try {
    let x = y; // y is not defined
} catch (error) {
    console.log("Error:", error.message);
} finally {
    console.log("Execution complete.");
}
